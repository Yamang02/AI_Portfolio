# P01: 스키마 재설계 및 타입 정의

## 목표

- `profile/src/types/profile.ts`가 `index.html`의 모든 콘텐츠 섹션(hero, identity, activities, interests)을 빠짐없이 표현한다.
- activities의 세 modal 레이아웃(`columns` / `site` / `none`)이 discriminated union으로 타입 안전하게 정의된다.
- 재정의된 타입이 TypeScript 컴파일 오류 없이 통과한다.

## 구현 상세

### 변경 대상

**`profile/src/types/profile.ts`** — 전체 재작성

### 섹션별 스키마 설계

#### hero

export interface ProfileHero {

#### identity (About Me + 역할 카드)

```ts
export interface ProfileIdentityCard {
  icon: string;         // Material Symbols 이름 e.g. "school"

```

#### activities (works — discriminated union)

```ts
// 공통 필드
interface ActivityBase {
  id: string;
  badge: string;
  cardTitle: string;
  cardTitleClass?: string;
  colSpan: number;
  minHeight: number;
}

// columns layout (국어교사, 개발경력)
export interface ColumnsActivity extends ActivityBase {
  modalLayout: 'columns';
  mediaType: 'img';
  mediaSrc: string;
  itemStyle?: 'career';           // default: teaching style
  columns: {
    label: string;
    items: {
      org: string;
      subOrg?: string;
      period: string;
      desc?: string;              // career style
      duties?: string[];          // teaching style
    }[];
  }[];
}

// site layout (artclub, design)
export interface SiteActivity extends ActivityBase {
  modalLayout: 'site';
  mediaType: 'img' | 'video';
  mediaSrc: string;
  period: string;
  body: string;
  link: string;
  linkLabel: string;
}

// none layout (highlight 카드, vibe)
export interface HighlightActivity extends ActivityBase {
  modalLayout: 'none';
  mediaType: 'none';
  highlight: true;
  body: string;                   // 상세 본문 (HTML 허용)
}

export type ProfileActivity = ColumnsActivity | SiteActivity | HighlightActivity;
```

#### interests

```ts
export interface ProfileInterest {
  icon: string;         // Material Symbols 이름 e.g. "explore"
  title: string;
  body: string;
}
```

#### 루트

```ts
export interface ProfileContent {
  hero: ProfileHero;
  identity: ProfileIdentitySection;
  activities: ProfileActivity[];
  interests: ProfileInterest[];
}
```
